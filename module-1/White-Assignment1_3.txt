def main():
    while True:

        #try/except clause to control for user input that is not an integer
        try:
            bottles = int(input("How many bottles of beer are left? Enter a number between 1 and 99: "))
            if bottles in range(1, 100):

                # exit while True loop if input is between 1 and 99
                break

            # if input is outside of range, prompt user to enter a number within the range
            print("Please enter a whole number between 1 and 99.")
        except ValueError:

            # display error message if user input is not an integer
            print("Please enter a whole number, as a numeric value, between 1 and 99.")

    # for loop controls countdown based on user input
    for count in range(bottles, 0, -1):
        if count > 2:
            print(f"{count} bottles of beer on the wall, {count} bottles of beer. "
                  f"Take one down and pass it around, {count - 1} bottles of beer on the wall.")

        # statement to control grammatical change when countdown goes from multiple bottles to one bottle
        elif count == 2:
            print("2 bottles of beer on the wall, 2 bottles of beer. "
                  "Take one down and pass it around, 1 bottle of beer on the wall.")
        # display a statement for when the countdown reaches the end
        else:
            print("1 bottle of beer on the wall, 1 bottle of beer. "
                  "Take one down and pass it around, 0 bottles of beer on the wall... Time to buy more beer!")

# call the main function
if __name__ == "__main__":
    main()
